VARIABLES

<?php
$variable = "Hello variable :)";
$variable2 = 'I\'m a variable'; //\escaping sign (pozwala użyć pojedyńczego '
$variable3 = "I am" . " a variable 3"; //zamiast + używamy kropki, żeby połączyć stringi
echo $variable;
echo $variable2;
echo $variable3;
?>
<?php
$age = 22;
echo "My age is $age "; //działa tylko z podwójnymi cudzysłowami" "" | jeżeli pojedyńcze, to trzeba użyć kropki .
$a = "age";
echo "My age is " . $$a . "<br>"; //robimy (przywołujemy) zmienną o nazwie zmiennej. (deklarujemy zmienną o nazwie$a. Ponieważ wcześniej już mamy zmienną o takiej nazwie, nasza nowa zmienna przyjmuje wartość 22. $$a => $age => 22).
?>
